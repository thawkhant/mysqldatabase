=>Full Join (UNION JOIN)


=Syntax

SELECT columns 
FROM table1 
LEFT JOIN table2
ON table1.column = table2.column
UNION
SELECT columns 
FROM table1 
RIGHT JOIN table2
ON table1.column = table2.column
