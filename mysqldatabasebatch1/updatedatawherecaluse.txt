==>UPDATE DATA BY USING CLAUSE
Notes:::Be caeful when updating records in a table;


UPDATE Syntax

UPDATE tablename 
SET column1=value1,column2=value2......
WHERE condition

Eg...

UPDATE coustomers
SET contactname="maung maung",city="mandalay"
WHERE customerid =1;

customerid        customername            city
1                 aung                    yangon
2                 maung maung             mandalay
